 36    def generate_next_state(self, nutrient_score):
 37        """
 38        Alter state with probability p. 
 39
 40        Args:
 41            nutrient_score (float): nutrient concentration in current cell.
 42        """
 43        probability_of_proliferate = self.probability_proliferate(nutrient_score)
 44        probability_of_invasion = self.probability_invasion(nutrient_score)
 45        normalized_proliferate = probability_of_proliferate / (probability_of_proliferate + probability_of_invasion)
 46
 47        random_value = np.random.random()
 48        if random_value < normalized_proliferate:
 49            self.next_state = 'proliferating'
 50        else:
 51            self.next_state = 'invasive'

 53    def p_proliferate_invasive(self, nutrient_score, which):
 54        """
 55        Return probability of the tumor cell be(com)ing proliferative/invasive.
 56
 57        Args:
 58            nutrient_score (float): nutrient concentration in current cell.
 59            which (str): which probability to return, invasive or proliferate.
 60        """
 61        assert which in ['invasive', 'proliferate'], 'which must be invasive or proliferate' 
 62        
 63        proliferate = (which == 'proliferate')
 64        theta = self.model.theta_p if proliferate else self.model.theta_i
 65
 66        # formula from paper by Chen et al., see https://www.nature.com/articles/srep17992 
 67        left = e**(-(nutrient_score / (self.model.N_T[self.pos] * theta))**2)
 68        left = 1 - left if proliferate else left
 69        
 70        right = 1
 71        neighboring_cells = self.model.grid.get_neighbors(self.pos, moore=True, include_center=True if proliferate else False) 
 72        
 73        # incorporate behavior of neighbors 
 74        for neighbor in neighboring_cells:
 75
 76            if neighbor.state == 'proliferating' and neighbor != self:
 77                right *= 1 + self.model.app if proliferate else 1 + self.model.bip
 78            elif neighbor.state == 'invasive' and neighbor != self: 
 79                right *= 1 + self.model.api if proliferate else 1 + self.model.bii
 80        
 81        return left * right

123        """
124        Create daughter cell and place on grid.

129        self.model.add_agent('proliferating', self.model.next_id(), best_cell)

132        """
133        State of tumor cell is set to necrotic and agent is removed.

135        self.state = 'necrotic'
136        self.model.grid.remove_agent(self)
137        self.remove()

139    def get_best_neighbor_site(self, nutrient_grid) -> tuple[int, int]:
140        """
141        Find optimal neighboring site for migration/daughter cell.
142
143        Sites with low tumor cells and high nutrient score are preferred.
144        Only grid cells with zero ECM (healthy tissue) are considered.
145
146        Returns:
147            tuple[int, int]: coordinates of best neighboring site.
148        """
149        # initial values
150        zero_ECM_sites = self.get_empty_ecm_sites()
151        open_cell = self.pos
152        lowest_amount = sys.maxsize
153
154        # loop over options
155        for ecm_0 in zero_ECM_sites:
156            number_of_agents = self.model.N_T[ecm_0] + self.model.Nec[ecm_0]
157
158            # prefer cells with low tumor density
159            if number_of_agents < lowest_amount:
160                lowest_amount = number_of_agents
161                open_cell = ecm_0
162
163            # if multiple have lowest density, choose on nutrients
164            elif number_of_agents == lowest_amount:
165                if nutrient_grid.data[open_cell] < nutrient_grid.data[ecm_0]:
166                    open_cell = ecm_0
167                elif nutrient_grid.data[open_cell] == nutrient_grid.data[ecm_0]:
168                    open_cell = random.choice((open_cell, ecm_0))
169
170        return open_cell
171    
172    def get_empty_ecm_sites(self) -> list[tuple[int, int]]:
173        """
174        Find all neighboring sites with zero ECM.
175
176        Returns:
177            list[tuple]: list of coordinates of neighboring zero ECM sites.
178        """
179        neighbors = self.model.grid.get_neighborhood(self.pos, moore=True, include_center=False)
180        zero_ECM_neighbors = []
181
182        for neighbor in neighbors:
183
184            if self.model.ecm_layer.data[neighbor] == 0:
185                zero_ECM_neighbors.append(neighbor)
186
187        return zero_ECM_neighbors
