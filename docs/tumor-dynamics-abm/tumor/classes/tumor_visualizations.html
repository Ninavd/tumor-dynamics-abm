 11    Example usage:

 23        """

 32    def show_ecm(self, position: int = -1):
 33        """
 34        Plot current ECM density field.
 35
 36        Args:
 37            position (int): which snapshot to plot. Default plots final snapshot.

 41        return fig, axs

 56        Plot mask of the tumor. Excludes necrotic cells.

 58        Args:

 65    def plot_necrotic_cells(self, position: int = -1):
 66        """
 67        Plot mask of the tumor, only including necrotic cells.
 68
 69        Args:
 70            position (int): which snapshot to plot. Default plots final snapshot.
 71        """
 72        fig, axs = plt.subplots()
 73        im = axs.imshow(self.model.Necs[position], cmap='BuPu')
 74        plt.title(f'Necrotic cell distribution for a {self.model.height}x{self.model.width} Grid at iteration {len(self.model.ecm_layers)-1}')
 75        plt.colorbar(im, fraction=0.046, pad=0.04)  #TODO: colorbar keeps getting errors
 76        return fig, axs

 78    def plot_tumor_over_time(self, steps: int):
 79        """
 80        Plot tumor over time.
 81
 82        Args:
 83            steps (int): Number of iterations performed in simulation.
 84        """
 85        final_fig, final_axs = plt.subplots(1, 3, figsize=(15, 5))
 86
 87        tumor_initial_fig, tumor_initial_axs = self.show_tumor(position = 0)
 88        tumor_middle_fig, tumor_middle_axs = self.show_tumor(position = round(steps/2))
 89        tumor_final_fig, tumor_final_axs = self.show_tumor(position = steps)
 90
 91        tumor_initial_axs = final_axs[0].imshow(tumor_initial_axs.get_images()[0].get_array(), vmin=0, vmax=5, cmap = 'BuPu')
 92        final_axs[0].axis("off")
 93        final_axs[0].set_title('t=0')
 94
 95        tumor_middle_axs = final_axs[1].imshow(tumor_middle_axs.get_images()[0].get_array(), vmin=0, vmax=5, cmap = 'BuPu')
 96        final_axs[1].set_title(f't={round(steps/2)}')
 97        final_axs[1].axis("off")
 98
 99        tumor_final_axs = final_axs[2].imshow(tumor_final_axs.get_images()[0].get_array(), vmin=0, vmax=5, cmap = 'BuPu')
100        final_axs[2].set_title(f't={steps}')
101        final_axs[2].axis("off")
102
103        plt.close(tumor_initial_fig)
104        plt.close(tumor_middle_fig)
105        plt.close(tumor_final_fig)
106        im_ratio = 1
107
108        final_fig.colorbar(tumor_final_axs, ax=final_axs.ravel().tolist(), fraction=0.046*im_ratio, pad=0.04)
109        plt.suptitle(f'Tumor Growth Over Time For a {self.model.height}x{self.model.width} Grid')
110        plt.show()

112    def plot_all(self, position: int = -1):
113        """
114        Plot ECM, nutrient field and tumour shape in a single figure.
115
116        Args:
117            position (int): which snapshot to plot. Default plots final snapshot.
118        """
119        if type(position) == int:
120            positions = [position]
121        else:
122            positions = position
123        
124        for position in positions:
125            if (position > len(self.model.ecm_layers) - 1):
126                print(f"Position {position} out of range. Max position: {len(self.model.ecm_layers) - 1}, skipping graphic iteration...")
127                continue
128            final_fig, final_axs = plt.subplots(1, 3, figsize=(15, 5))
129
130            ecm_fig, ecm_axs = self.show_ecm(position = position)
131            nutrient_fig, nutrient_axs = self.show_nutrients(position = position)
132            tumor_fig, tumor_axs = self.show_tumor(position = position)
133
134            ecm_axs = final_axs[0].imshow(ecm_axs.get_images()[0].get_array(), vmin = 0, vmax = 1, cmap = 'BuPu')
135            final_axs[0].axis("off")
136            final_axs[0].set_title('ECM Field Concentration')
137
138            nutrient_axs = final_axs[1].imshow(nutrient_axs.get_images()[0].get_array(), cmap = 'BuPu')
139            final_axs[1].set_title('Nutrient Field Concentration')
140            final_axs[1].axis("off")
141
142            tumor_axs = final_axs[2].imshow(tumor_axs.get_images()[0].get_array(), cmap = 'BuPu')
143            final_axs[2].set_title('Tumor Cell Count')
144            final_axs[2].axis("off")
145
146            plt.close(ecm_fig)
147            plt.close(nutrient_fig)
148            plt.close(tumor_fig)
149
150            final_fig.colorbar(ecm_axs, ax=final_axs[0], fraction=0.046, pad=0.04)
151            final_fig.colorbar(nutrient_axs, ax=final_axs[1], fraction=0.046, pad=0.04)
152            final_fig.colorbar(tumor_axs, ax=final_axs[2], fraction=0.046, pad=0.04)
153            plt.suptitle(f'ECM, Nutrient, and Tumor Values at Iteration {position%self.model.steps + 1} of {self.model.steps} for a {self.model.height}x{self.model.width} Grid')
154            plt.show()

156    def plot_birth_deaths(self):
157        """
158        Plot cumulative number of births and deaths.
159        """
160        birth_rel_death = [(self.model.births[i]) /(self.model.births[i] + self.model.necrotic_cells[i]) for i in range(len(self.model.births))]
161        fig, ax1 = plt.subplots()
162        ax2 = ax1.twinx()
163        ax1.plot(self.model.births, label='Births')
164        ax1.plot(self.model.necrotic_cells, label='Deaths')
165        ax2.plot(birth_rel_death, label='Births / (Births + Deaths)', color='g')
166
167        ax1.set_xlabel('Iteration')
168        ax1.set_ylabel('Count')
169        ax2.set_ylabel('Relative fracion', color='g')
170        ax2.tick_params(colors='green', which='both')
171        ax2.set_ylim(0, 1.05)
172        handles1, labels1 = ax1.get_legend_handles_labels()
173        handles2, labels2 = ax2.get_legend_handles_labels()
174        plt.legend(handles1 + handles2, labels1 + labels2)
175        plt.title('Cumulative Number of Births and Deaths')
176        plt.show()
